.data
	tb1: .asciiz "Nhap n: "
	tb2: .asciiz "Tong 1 den n la: "

.text
	#xuat tb1
	l1 $v0, 4
	la $a0, tb1
	syscall

	# nhap n
	li $v0, 5
	syscall
	
	#luu vao s0
	move $s0, $v0

	#khoi tao
	li $s1, 0 # s = 0
	li $t0, 1 # i = 1
Lap:
	add $s1, $s1, $t0 #s = s + i
	addi $t0, $t0, 1 #i = i + 1

	# kiem tra i <= n thi lap
	# slt $t1, $s0, $t0
	# beq $t1, 0, Lap
	ble $t0, $s0, Lap

	# xuat tb2
	li $v0, 4
	la $a0, tb2
	syscall

	# xuat s1
	li $v0, 1
	move $a0, $s1
	syscall
